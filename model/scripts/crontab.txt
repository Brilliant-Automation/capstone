# Cron job for running the data pipeline
# This file documents the cron schedule but should not be installed directly
# To install, run: crontab -e and add the following line with your specific path

# Run pipeline every hour
0 * * * * /<absolute_path>/model/scripts/run_pipeline.sh

# Note: Replace <absolute_path> with your actual project path
# For example, if your project is at /home/jupyter-ubuntu/capstone,
# the full path would be:
# 0 * * * * /home/jupyter-ubuntu/capstone/model/scripts/run_pipeline.sh

# To view current cron jobs:
# crontab -l

# To edit cron jobs:
# crontab -e

# To remove all cron jobs:
# crontab -r
